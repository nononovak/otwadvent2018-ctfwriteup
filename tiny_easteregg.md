def hex_dna(dna):
	pats = []
	for ch1 in 'ATGC':
		for ch2 in 'ATGC':
			if ch1 == ch2:
				continue
			for ch3 in 'ATGC':
				if ch1 == ch3 or ch2 == ch3:
					continue
				for ch4 in 'ATGC':
					if ch1 == ch4 or ch2 == ch4 or ch3 == ch4:
						continue
					pats.append(ch1+ch2+ch3+ch4)
	print(pats)

	dna = ''.join(dna.strip().split())
	for pat in pats:
		arr = []
		for i in range(0,len(dna),4):
			x1 = pat.index(dna[i])
			x2 = pat.index(dna[i+1])
			x3 = pat.index(dna[i+2])
			x4 = pat.index(dna[i+3])
			x = (x1<<6)|(x2<<4)|(x3<<2)|x4
			arr.append(x)
		buf = bytes(arr)
		with open('tiny/dna_%s.txt'%pat,'wb') as f:
			f.write(buf)
			f.close()
		print(pat, buf)

def dna_hex(buf):

	for b in buf:

		st.append(pat[(b>>2)&3])
		st.append(pat[(b>>0)&3])
	st = ''.join(st)

	return st
